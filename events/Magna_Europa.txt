# --- Option A: Annex + Cores + 2 Civ
country_event = {
    id = MA_germany.1a
    title = MA_germany.1a.t
    desc  = MA_germany.1a.desc
    is_triggered_only = yes

    immediate = {
        annex_country = { target = LIE transfer_troops = yes }

        every_state = {
            limit = { is_core_of = LIE }
            add_core_of = GER
        }

        # In State 1460 eine Civ sofort bauen
        1460 = {
			add_extra_state_shared_building_slots = 2
            add_building_construction = {
                type = industrial_complex
                level = 2
                instant_build = yes
            }
        }
    }
}
